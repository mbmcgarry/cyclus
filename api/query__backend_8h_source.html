00373 /// The digest type for SHA1s.
00374 ///
00375 /// This class is a hack around a language deficiency in C++. You cannot pass
00376 /// around an array (unsinged int[5]) between function calls. You can only
00377 /// pass pointers, which would involve lost of new/free and heap shenanigans
00378 /// that are not needed for a dumb container. Therefore Sha1::Digest() cannot
00379 /// return what would be most natural. The second most natural thing would be
00380 /// a std::array<unsigned int, 5>. However, std::array is a C++11 feature and
00381 /// we are not yet ready to go down that road.
00382 ///
00383 /// To pass an array into and out of a function it has to be inside of struct
00384 /// or a class. I chose a class here since there are many member functions.
00385 ///
00386 /// The reason why this is public is that it needs to be directly writable
00387 /// from buffers coming from HDF5. In the future, this really should just be
00388 /// a std::array.
00389 class Digest {
00390  public:
00391   unsigned int val[CYCLUS_SHA1_NINT];
00392 
00393   /// Casts the value of this digest to a vector of the templated type.
00394   template <typename T>
00395   inline std::vector<T> cast() const {
00396     std::vector<T> rtn = std::vector<T>(CYCLUS_SHA1_NINT);
00397     for (unsigned int i = 0; i < CYCLUS_SHA1_NINT; ++i)
00398       rtn[i] = static_cast<T>(val[i]);
00399     return rtn;
00400   }
00401 
00402   // operators
00403   inline std::ostream& operator<<(std::ostream& out) const {
00404     return out << "[" << val[0] << ", " << val[1] << ", " <<  val[2] << \
00405                   ", " << val[3] << ", " << val[4] << "]";
00406   }
00407 
00408   inline bool operator< (const cyclus::Digest& rhs) const {
00409     bool rtn = false;
00410     for (int i = 0; i < CYCLUS_SHA1_NINT; ++i) {
00411       if (val[i] < rhs.val[i]) {
00412         rtn = true;
00413         break;
00414       } else if (val[i] > rhs.val[i]) {
00415         rtn = false;
00416         break;
00417       }  // else they are equal and we need to check the next index
00418     }
00419     return rtn;
00420   }
00421 
00422   inline bool operator> (const cyclus::Digest& rhs) const {
00423     return !operator<(rhs) && !operator==(rhs);
00424   }
00425 
00426   inline bool operator<=(const cyclus::Digest& rhs) const {
00427     return !operator>(rhs);
00428   }
00429 
00430   inline bool operator>=(const cyclus::Digest& rhs) const {
00431     return !operator<(rhs);
00432   }
00433 
00434   inline bool operator==(const cyclus::Digest& rhs) const {
00435     bool rtn = true;
00436     for (int i = 0; i < CYCLUS_SHA1_NINT; ++i) {
00437       if (val[i] != rhs.val[i]) {
00438         rtn = false;
00439         break;
00440       }  // else they are equal and we need to check the next index.
00441     }
00442     return rtn;
00443   }

00445   inline bool operator!=(const cyclus::Digest& rhs) const {
