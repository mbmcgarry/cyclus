00023 /// The material class is primarily responsible for enabling basic material
00024 /// manipulation while helping enforce mass conservation.  It also provides the
00025 /// ability to easily decay a material up to the current simulation time; it
00026 /// does not perform any decay related logic itself.
00027 ///
00028 /// There are four basic operations that can be performed on materials: create,
00029 /// transmute (change material composition - e.g. fission by reactor), absorb
00030 /// (combine materials), extract (split a material). All material
00031 /// handling/manipulation will be performed using these operations - and all
00032 /// operations performed will be tracked and recorded. Usage examples:
00033 ///
00034 /// * A mining facility that "creates" new material
00035 ///
00036 ///   @code
00037 ///   Composition::Ptr nat_u = ...
00038 ///   double qty = 10.0;
00039 ///
00040 ///   Material::Ptr m = Material::Create(qty, nat_u, ctx);
00041 ///   @endcode
00042 ///
00043 /// * A conversion facility mixing uranium and flourine:
00044 ///
00045 ///   @code
00046 ///   Material::Ptr uf6 = uranium_buf.Pop();
00047 ///   Material::Ptr f = flourine_buf.Pop();
00048 ///
00049 ///   uf6.Absorb(f);
00050 ///   @endcode
00051 ///
00052 /// * A reactor transmuting fuel:
00053 ///
00054 ///   @code
00055 ///   Composition::Ptr burned_comp = ... // fancy code to calculate burned nuclides
00056 ///   Material::Ptr assembly = core_fuel.Pop();
00057 ///
00058 ///   assembly.Transmute(burned_comp);
00059 ///   @endcode
00060 ///
00061 /// * A separations plant extracting stuff from spent fuel:
00062 ///
00063 ///   @code
00064 ///   Composition::Ptr comp = ... // fancy code to calculate extracted nuclides
00065 ///   Material::Ptr bucket = spent_fuel.Pop();
00066 ///   double qty = 3.0;
00067 ///
00068 ///   Material::Ptr mox = bucket.ExtractComp(qty, comp);
00069 ///   @endcode
00070 ///
