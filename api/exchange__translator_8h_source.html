00222 /// @brief translates an arc given a bid and subsequent data, and also
00223 /// updates the unit capacities for the associated nodes on the arc
00224 template <class T>
00225 Arc TranslateArc(const ExchangeTranslationContext<T>& translation_ctx,
00226                  Bid<T>* bid) {
00227   Request<T>* req = bid->request();
00228 
00229   ExchangeNode::Ptr unode = translation_ctx.request_to_node.at(req);
00230   ExchangeNode::Ptr vnode = translation_ctx.bid_to_node.at(bid);
00231   Arc arc(unode, vnode);
00232 
00233   typename T::Ptr offer = bid->offer();
00234   typename BidPortfolio<T>::Ptr bp = bid->portfolio();
00235   typename RequestPortfolio<T>::Ptr rp = req->portfolio();
00236 
00237   // bid is v
00238   TranslateCapacities(offer, bp->constraints(), vnode, arc, translation_ctx);
00239   // req is u
00240   TranslateCapacities(offer, rp->constraints(), unode, arc, translation_ctx);
