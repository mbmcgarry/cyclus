01441         if (vldts_.count(VL_MAP_STRING_STRING) == 0) {
01442           vldts_[VL_MAP_STRING_STRING] = H5Tvlen_create(item_type);
01443           opened_types_.insert(vldts_[VL_MAP_STRING_STRING]);
01444         } else {
01445           H5Tclose(item_type);
01446         }
01447         dst_sizes[i] = CYCLUS_SHA1_SIZE;
01448       } else if (shape[0] >= 1 && shape[1] < 1 && shape[2] < 1) {
01449         hsize_t shape0 = shape[0];
01450         dbtypes[i] = MAP_VL_STRING_VL_STRING;
01451         item_type = H5Tcreate(H5T_COMPOUND, 2*CYCLUS_SHA1_SIZE);
01452         H5Tinsert(item_type, "key", 0, sha1_type_);
01453         H5Tinsert(item_type, "val", CYCLUS_SHA1_SIZE, sha1_type_);
01454         field_types[i] = H5Tarray_create2(item_type, 1, &shape0);
01455         opened_types_.insert(item_type);
01456         opened_types_.insert(field_types[i]);
01457         dst_sizes[i] = shape[0] * 2 * CYCLUS_SHA1_SIZE;
01458       } else if (shape[0] >= 1 && shape[1] < 1 && shape[2] >= 1) {
01459         hsize_t shape0 = shape[0];
01460         dbtypes[i] = MAP_VL_STRING_STRING;
01461         hid_t val_type = CreateFLStrType(shape[2]);
01462         item_type = H5Tcreate(H5T_COMPOUND, CYCLUS_SHA1_SIZE + shape[2]);
01463         H5Tinsert(item_type, "key", 0, sha1_type_);
01464         H5Tinsert(item_type, "val", CYCLUS_SHA1_SIZE, val_type);
01465         field_types[i] = H5Tarray_create2(item_type, 1, &shape0);
01466         opened_types_.insert(item_type);
01467         opened_types_.insert(field_types[i]);
01468         dst_sizes[i] = shape[0] * (CYCLUS_SHA1_SIZE + shape[2]);
01469       } else if (shape[0] >= 1 && shape[1] >= 1 && shape[2] < 1) {
01470         hsize_t shape0 = shape[0];
01471         dbtypes[i] = MAP_STRING_VL_STRING;
01472         hid_t key_type = CreateFLStrType(shape[1]);
01473         item_type = H5Tcreate(H5T_COMPOUND, CYCLUS_SHA1_SIZE + shape[1]);
01474         H5Tinsert(item_type, "key", 0, key_type);
01475         H5Tinsert(item_type, "val", shape[1], sha1_type_);
01476         field_types[i] = H5Tarray_create2(item_type, 1, &shape0);
01477         opened_types_.insert(item_type);
01478         opened_types_.insert(field_types[i]);
01479         dst_sizes[i] = shape[0] * (CYCLUS_SHA1_SIZE + shape[1]);
01480       } else {
01481         dbtypes[i] = MAP_STRING_STRING;
01482         hsize_t shape0 = shape[0];
01483         hid_t key_type = CreateFLStrType(shape[1]);
01484         hid_t val_type = CreateFLStrType(shape[2]);
01485         item_type = H5Tcreate(H5T_COMPOUND, shape[1] + shape[2]);
01486         H5Tinsert(item_type, "key", 0, key_type);
01487         H5Tinsert(item_type, "val", shape[1], val_type);
01488         field_types[i] = H5Tarray_create2(item_type, 1, &shape0);
01489         opened_types_.insert(item_type);
01490         opened_types_.insert(field_types[i]);
01491         dst_sizes[i] = shape[0] * (shape[1] + shape[2]);
01492       }
01493     } else {
01494       throw IOError("the type for column '" + std::string(field_names[i]) + \
01495                     "' is not yet supported in HDF5.");
01496     }
01497     dst_size += dst_sizes[i];
01498   }
01499 
01500   const char* title = d->title().c_str();
01501   int compress = 1;
01502   int chunk_size = 1024;
01503   void* fill_data = NULL;
01504   void* data = NULL;
01505 
01506   // Make the table
01507   status = H5TBmake_table(title, file_, title, nvals, 0, dst_size,
01508                           field_names, dst_offset, field_types, chunk_size,
01509                           fill_data, compress, data);
01510   if (status < 0) {
01511     std::stringstream ss;
01512     ss << "Failed to create HDF5 table:\n" \
01513        << "  file      " << path_ << "\n" \
01514        << "  table     " << title << "\n" \
01515        << "  chunksize " << chunk_size << "\n" \
01516        << "  rowsize   " << dst_size << "\n";
01517     for (int i = 0; i < nvals; ++i) {
01518       ss << "    #" << i << " " << field_names[i] << "\n" \
01519          << "      dbtype: " << dbtypes[i] << "\n" \
01520          << "      h5type: " << field_types[i] << "\n" \
01521          << "      size:   " << dst_sizes[i] << "\n" \
01522          << "      offset: " << dst_offset[i] << "\n";
01523     }
01524     throw IOError(ss.str());
01525   }
01526 
01527   // add dbtypes attribute
01528   hid_t tb_set = H5Dopen2(file_, title, H5P_DEFAULT);
01529   hid_t attr_space = H5Screate_simple(1, &nvals, &nvals);
01530   hid_t dbtypes_attr = H5Acreate2(tb_set, "cyclus_dbtypes", H5T_NATIVE_INT,
01531                                   attr_space, H5P_DEFAULT, H5P_DEFAULT);
01532   H5Awrite(dbtypes_attr, H5T_NATIVE_INT, dbtypes);
01533   H5Aclose(dbtypes_attr);
01534   H5Sclose(attr_space);
01535   H5Dclose(tb_set);
01536 
01537   // record everything for later
01538   col_offsets_[d->title()] = dst_offset;
01539   schema_sizes_[d->title()] = dst_size;
01540   col_sizes_[d->title()] = dst_sizes;
01541   schemas_[d->title()] = dbtypes;
01542 }

01544 void Hdf5Back::WriteGroup(DatumList& group) {
01545   std::string title = group.front()->title();
01546   const char * c_title = title.c_str();
01547 
01548   size_t* offsets = col_offsets_[title];
01549   size_t* sizes = col_sizes_[title];
01550   size_t rowsize = schema_sizes_[title];
01551 
01552   char* buf = new char[group.size() * rowsize];
01553   FillBuf(title, buf, group, sizes, rowsize);
01554 
01555   // We cannot do the simple thing (append_records) here because of a bug in
01556   // H5TB where it stupidly tries to reconstruct the datatype in memory from
01557   // what it read in on disk. This works in most cases but failed where the table
01558   // had a column which is an array of a compound datatype of non-homogenous
01559   // fields (eg MAP_INT_DOUBLE). The fix here just uses the datatype present on
01560   // disk - which is what we wanted anyway!
01561   //herr_t status = H5TBappend_records(file_, title.c_str(), group.size(), rowsize,
01562   //                            offsets, sizes, buf);
01563   herr_t status;
01564   hid_t dset = H5Dopen2(file_, title.c_str(), H5P_DEFAULT);
01565   hid_t dtype = H5Dget_type(dset);
01566   hsize_t nrecords_add = group.size();
01567   hsize_t nrecords_orig;
01568   hsize_t nfields;
01569   hsize_t dims[1];
01570   hsize_t offset[1];
01571   hsize_t count[1];
01572   H5TBget_table_info(file_, c_title, &nfields, &nrecords_orig);
01573   dims[0] = nrecords_add + nrecords_orig;
01574   offset[0] = nrecords_orig;
01575   count[0] = nrecords_add;
01576 
01577   status = H5Dset_extent(dset, dims);
01578   hid_t dspace = H5Dget_space(dset);
01579   hid_t memspace = H5Screate_simple(1, count, NULL);
01580   status = H5Sselect_hyperslab(dspace, H5S_SELECT_SET, offset, NULL, count, NULL);
01581   status = H5Dwrite(dset, dtype, memspace, dspace, H5P_DEFAULT, buf);
01582 
01583   if (status < 0) {
01584     std::stringstream ss;
01585     ss << "Failed to write to the HDF5 table:\n" \
01586        << "  file      " << path_ << "\n" \
01587        << "  table     " << title << "\n" \
01588        << "  num. rows " << group.size() << "\n"
01589        << "  rowsize   " << rowsize << "\n";
01590     for (int i = 0; i < group.front()->vals().size(); ++i) {
01591       ss << "    # Column " << i << "\n" \
01592          << "      dbtype: " << schemas_[title][i] << "\n" \
01593          << "      size:   " << sizes[i] << "\n" \
01594          << "      offset: " << offsets[i] << "\n";
01595     }
01596     throw IOError(ss.str());
01597   }
01598 
01599   H5Sclose(memspace);
01600   H5Sclose(dspace);
01601   H5Tclose(dtype);
01602   H5Dclose(dset);
01603   delete[] buf;
01604 }

01649 void Hdf5Back::FillBuf(std::string title, char* buf, DatumList& group,
01650                        size_t* sizes, size_t rowsize) {

01652   using std::string;
01653   using std::vector;
01654   using std::set;
01655   using std::list;
01656   using std::pair;
01657   using std::map;
01658   Datum::Vals vals;
01659   Datum::Shape shape;
01660   Datum::Shapes shapes;
01661   Datum::Vals header = group.front()->vals();
01662   int ncols = header.size();
01663   DbTypes* dbtypes = schemas_[title];
01664 
01665   size_t offset = 0;
01666   const void* val;
01667   size_t fieldlen;
01668   size_t valuelen;
01669   DatumList::iterator it;
01670   for (it = group.begin(); it != group.end(); ++it) {
01671     vals = (*it)->vals();
01672     shapes = (*it)->shapes();
01673     for (int col = 0; col < ncols; ++col) {
01674       const boost::spirit::hold_any* a = &(vals[col].second);
01675       switch (dbtypes[col]) {
01676         case BOOL:
01677         case INT:
01678         case FLOAT:
01679         case DOUBLE: {
01680           val = a->castsmallvoid();
01681           memcpy(buf + offset, val, sizes[col]);
01682           break;
01683         }
01684         case STRING: {
01685           const std::string s = a->cast<std::string>();
01686           fieldlen = sizes[col];
01687           valuelen = std::min(s.size(), fieldlen);
01688           memcpy(buf + offset, s.c_str(), valuelen);
01689           memset(buf + offset + valuelen, 0, fieldlen - valuelen);
01690           break;
01691         }
01692         case VL_STRING: {
01693           Digest key = VLWrite<std::string, VL_STRING>(a);
01694           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01695           break;
01696         }
01697         case BLOB: {
01698           Digest key = VLWrite<Blob, BLOB>(a);
01699           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01700           break;
01701         }
01702         case UUID: {
01703           boost::uuids::uuid uuid = a->cast<boost::uuids::uuid>();
01704           memcpy(buf + offset, &uuid, CYCLUS_UUID_SIZE);
01705           break;
01706         }
01707         case VECTOR_INT: {
01708           std::vector<int> val = a->cast<std::vector<int> >();
01709           fieldlen = sizes[col];
01710           valuelen = std::min(val.size() * sizeof(int), fieldlen);
01711           memcpy(buf + offset, &val[0], valuelen);
01712           memset(buf + offset + valuelen, 0, fieldlen - valuelen);
01713           break;
01714         }
01715         case VL_VECTOR_INT: {
01716           Digest key = VLWrite<std::vector<int>, VL_VECTOR_INT>(a);
01717           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01718           break;
01719         }
01720         case VECTOR_FLOAT: {
01721           std::vector<float> val = a->cast<std::vector<float> >();
01722           fieldlen = sizes[col];
01723           valuelen = std::min(val.size() * sizeof(float), fieldlen);
01724           memcpy(buf + offset, &val[0], valuelen);
01725           memset(buf + offset + valuelen, 0, fieldlen - valuelen);
01726           break;
01727         }
01728         case VL_VECTOR_FLOAT: {
01729           Digest key = VLWrite<std::vector<float>, VL_VECTOR_FLOAT>(a);
01730           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);

01732         }
01733         case VECTOR_DOUBLE: {
01734           std::vector<double> val = a->cast<std::vector<double> >();
01735           fieldlen = sizes[col];
01736           valuelen = std::min(val.size() * sizeof(double), fieldlen);
01737           memcpy(buf + offset, &val[0], valuelen);
01738           memset(buf + offset + valuelen, 0, fieldlen - valuelen);
01739           break;
01740         }
01741         case VL_VECTOR_DOUBLE: {
01742           Digest key = VLWrite<std::vector<double>, VL_VECTOR_DOUBLE>(a);
01743           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01744           break;
01745         }
01746         case VECTOR_STRING: {

01749           fieldlen = shape[1];

01751           for (; cnt < val.size(); ++cnt) {
01752             valuelen = std::min(val[cnt].size(), fieldlen);
01753             memcpy(buf + offset + fieldlen*cnt, val[cnt].c_str(), valuelen);
01754             memset(buf + offset + fieldlen*cnt + valuelen, 0, fieldlen - valuelen);
01755           }
01756           memset(buf + offset + fieldlen*cnt, 0, fieldlen * (shape[0] - cnt));
01757           break;
01758         }
01759         case VECTOR_VL_STRING: {
01760           vector<string> val = a->cast<vector<string> >();

01762           unsigned int cnt = 0;

01770         }
01771         case VL_VECTOR_STRING: {

01776           unsigned int cnt = 0;

01783         }

01851           fieldlen = sizes[col];

01853           unsigned int cnt = 0;
01854           std::list<int>::iterator valit = val.begin();
01855           for (; valit != val.end(); ++valit) {
01856             memcpy(buf + offset + cnt*sizeof(int), &(*valit), sizeof(int));
01857             ++cnt;
01858           }
01859           memset(buf + offset + valuelen, 0, fieldlen - valuelen);
01860           break;
01861         }
01862         case VL_LIST_INT: {
01863           Digest key = VLWrite<std::list<int>, VL_LIST_INT>(a);
01864           memcpy(buf + offset, key.val, CYCLUS_SHA1_SIZE);
01865           break;
01866         }
01867         case LIST_STRING: {
01868           list<string> val = a->cast<list<string> >();
01869           shape = shapes[col];
01870           fieldlen = shape[1];
01871           unsigned int cnt = 0;
01872           list<string>::iterator valit = val.begin();
01873           for (; valit != val.end(); ++valit) {
01874             valuelen = std::min(valit->size(), fieldlen);
